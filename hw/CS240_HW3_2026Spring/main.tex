\textbf{Notice: When proving that a problem $A$ is NP-complete, you need to strictly follow the steps below and explicitly state each part.}

\begin{itemize}
	\item Show that $A\in \mathrm{NP}$.
	
	That is, given a candidate solution, we can verify it in polynomial time.
	
	\item Choose a known NP-complete problem $B$.
	
	\item Give a polynomial-time reduction from $B$ to $A$.
	
	For any instance $b$ of $B$, construct an instance $a$ of $A$ in polynomial time.
	
	\item Prove correctness of the reduction.
	
	Show that
	\[
	b \text{ is a YES-instance of } B
	\quad\Longleftrightarrow\quad
	a \text{ is a YES-instance of } A.
	\]
\end{itemize}
