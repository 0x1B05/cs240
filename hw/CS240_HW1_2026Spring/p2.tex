\problem{}
Determine the time complexity of the pseudocode below and explain your reasoning.

\begin{lstlisting}
def count_triples(arr, threshold):
    n = len(arr)
    count1 = 0
    count2 = 0

    def merge_sort(arr):
        if len(arr) <= 1:
            return arr
        
        mid = len(arr) // 2
        left = merge_sort(arr[:mid])
        right = merge_sort(arr[mid:])
        
        result = []
        i = j = 0
        
        while i < len(left) and j < len(right):
            if left[i] <= right[j]:
                result.append(left[i])
                i += 1
            else:
                result.append(right[j])
                j += 1
        
        result.extend(left[i:])
        result.extend(right[j:])
        
        return result

    arr = merge_sort(arr)

    for i in range(n):
        for j in range(i + 1, n):
            k = j + 1
            while k < n and arr[i] + arr[j] + arr[k] <= threshold:
                count1 += 1
                k += 1

    for i in range(n):
        left = i + 1
        right = n - 1
        
        while left < right:
            if arr[i] + arr[left] + arr[right] <= threshold:
                count2 += (right - left)
                left += 1
            else:
                right -= 1

    return count1 + count2
\end{lstlisting}
